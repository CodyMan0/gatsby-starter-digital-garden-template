```js
function solution(rsp) {
    const result = rsp.split('').map((element) => (
        element === '2' ? '0' : element === '5' ? '2' : '5'))

    return result.join('');
}
```


```js

function solution(rsp) {
    let arr = {
        2: 0,
        0: 5,
        5: 2
    };
    var answer = [...rsp].map(v => arr[v]).join("");
    return answer;
}
```
